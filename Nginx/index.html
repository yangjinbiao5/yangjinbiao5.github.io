常见用途

  1. web服务器 
    
server {  
       listen       80;   # 监听端口
       server_name  localhost; # 绑定域名
       location / {  
           root   /usr/share/nginx/html;  
           index  index.html index.htm;  
       }
}


server {  
       listen       443 ssl;  # 监听端口
       server_name  localhost; # 绑定域名
       ssl_certificate      cert.pem;
       ssl_certificate_key  cert.key;
       location / {  
           root   /usr/share/nginx/html;  
           index  index.html index.htm;  
       }
}
  2. 反向代理
    
server {  
       listen       80;  # 监听端口
       server_name  localhost; # 绑定域名
       location / {  
	       proxy_pass http://localhost:8080; # 后端服务器与端口
	       proxy_set_header Host $host; # 保留原始Host头
	       proxy_set_header X-Real-IP $remote_addr; # 传递真实客户IP
       }
       
}
  3. 负载均衡
    upstream webservers{
	
	server 192.168.1.100:8080;
	server 192.168.1.101:8080;
	server 192.168.1.102:8080;
	# 可选的负载均衡策略 默认轮询
	# ip_hash; # IP哈希
	# least_conn; # 最少连接
}
 server {  
       listen       80;  # 监听端口
       server_name  localhost; # 绑定域名
       location / {  
	       proxy_pass http://webservers;
	       proxy_set_header Host $host; # 保留原始Host头
	       proxy_set_header X-Real-IP $remote_addr; # 传递真实客户IP
       }
       
}
  4. URL重定向
     # 将老域名重定向到新域名 保持url不变
 server {  
       listen       80;  # 监听端口
       server_name  old.com; # 绑定域名
       location / {  
	       rewrite ^/(.*)$ https://new.com/$1;
       }
       
}
# URL变更
server {  
       listen       80;  # 监听端口
       server_name  localhost; # 绑定域名
       location / {  
	       rewrite ^/old-path/(.*)$ /new-path/$1;
       }
       
}
  5. 防盗链
    server {  
       listen       80;  # 监听端口
       server_name  localhost; # 绑定域名
       location ~* \.(gif|jpg|png)$ {  
	       valid_referers none blocked *.something.com;
	       if($invalid_referer) {
		       return 403;
	       }
       }
       
}
  6. 简易文件服务器
    server {  
       listen       80;  # 监听端口
       server_name  localhost; # 绑定域名
       # 完整路径 /home/cartman/Downloads/test
       location /test {  
	       root /home/cartman/Downloads;  
	       # 显示目录;    
		   autoindex on;  
		   # 显示文件大小; 关闭计算文件确切大小（单位bytes），只显示大概大小（单位kb、mb、gb)    
		   autoindex_exact_size off;    
		   # 显示文件时间    
		   autoindex_localtime on;
		   # 避免中文乱码;    
		   charset utf-8;
       }
}
  7. 基于不同请求路径转发不同服务
    server {  
       listen       80;  # 监听端口
       server_name  localhost; # 绑定域名
       location /path1/ {  
	       proxy_pass http://ip1:8080; # 后端服务器与端口
	       proxy_set_header Host $host; # 保留原始Host头
	       proxy_set_header X-Real-IP $remote_addr; # 传递真实客户IP
       }
       location /path2/ {  
	       proxy_pass http://ip2:8080; # 后端服务器与端口
	       proxy_set_header Host $host; # 保留原始Host头
	       proxy_set_header X-Real-IP $remote_addr; # 传递真实客户IP
       }
       location /path3/ {  
	       proxy_pass http://ip3:8080; # 后端服务器与端口
	       proxy_set_header Host $host; # 保留原始Host头
	       proxy_set_header X-Real-IP $remote_addr; # 传递真实客户IP
       }
       
}
